verwalter.useMiddleware();
public useMiddleware(): void {
	const infoLogger = 
	(req: Request, res: Response, next: NextFunction) 
	=> {
		console.log(`PrototypBackend: A ${req.method}
		-request on ${req.path} was made by ${req.ip}`);
		next();
	};
	this.app.use(infoLogger);
	this.app.use(express.json());
	this.app.use(cors());
}

}

private addPreorder(req: Request, res: Response) {
	let userId = req.body.userId;
	let orderItems = req.body.orderItems;
	let newOrder = 
		new Order(
			Math.floor(Math.random() * 10000), 
			userId, 
			orderItems
		);
	this.orders = [newOrder, ...this.orders]
	res.status(204);
	res.send('ok');
}
